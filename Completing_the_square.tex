\begin{enumerate}
\item
\[
x^2 -2x + 6
\]
\[
= (x-1)^2 + 5.
\]
Hence 
\[
x^2-2x+6 \geq 5.
\]
Because $(x-1)^2$ and therefore $x^2 -2x + 6$ is increasing in $[2,3]$ the minimum value it attains is attained in this interval when $x=2$ and so $x^2 -2x+6 = 6$. 
\item
\[
x^4 + 2x^2 + 2
\]
\[
= (x^2 +1)^2 + 1.
\]
We know that 
\[
x^2 \geq 0
\]
and consequently
\[
x^2 + 1 \geq 1
\]
and so
\[
(x^2 + 1)^2 \geq 1
\]
and hence, finally,
\[
x^4 + 2x^2 + 2 \geq 2.
\]
\end{enumerate}
