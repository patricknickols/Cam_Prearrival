\begin{enumerate}
\item
\[
x^2 - 5x + 6 = 0
\]
\[
\iff (x-3)(x-2) = 0
\]
\[
\iff x = 2,3.
\]
\item
\[
x^2 + 2x = 0
\]
\[
\iff (x)(x+2) = 0
\]
\[
\iff x = -2, 0.
\]
\item
\[
x^2 - x -1 = 0.
\]
\[
\iff x = \frac{1\pm\sqrt{5}}{2}.
\]
\item
\[
x^4 - 3x^2 + 2 = 0.
\]
We let $y = x^2$ so
\[
y^2 -3y + 2 = 0
\]
\[
\iff (y-2)(y-1) = 0,
\]
and so
\[
y = 1,2
\]
which means
\[
x = \pm 1, \pm \sqrt{2}.
\]
\end{enumerate}
